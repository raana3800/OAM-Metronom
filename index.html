import React, { useEffect, useRef, useState } from "react";

import { Card, CardContent } from "@/components/ui/card";

import { Button } from "@/components/ui/button";

import { Music, Square, Play } from "lucide-react";


const SONGS = [

  { title: "James Blunt", bpm: 105 },

  { title: "Paramore", bpm: 70 },

  { title: "Bruno Mars", bpm: 150 },

];


export default function ThreeSongMetronomeApp() {

  const audioContextRef = useRef(null);

  const intervalRef = useRef(null);

  const [activeSong, setActiveSong] = useState(null);

  const [isPlaying, setIsPlaying] = useState(false);

  const [flash, setFlash] = useState(false);


  const ensureAudioContext = async () => {

    if (!audioContextRef.current) {

      const AudioContextClass = window.AudioContext || window.webkitAudioContext;

      audioContextRef.current = new AudioContextClass();

    }


    if (audioContextRef.current.state === "suspended") {

      await audioContextRef.current.resume();

    }


    return audioContextRef.current;

  };


  // 🔥 besserer, knackiger Klick

  const clickSound = async () => {

    const ctx = await ensureAudioContext();

    const now = ctx.currentTime;


    const oscillator = ctx.createOscillator();

    const gain = ctx.createGain();


    oscillator.type = "square";

    oscillator.frequency.value = 1800; // setzt sich besser durch


    gain.gain.setValueAtTime(0.0001, now);

    gain.gain.exponentialRampToValueAtTime(0.25, now + 0.002);

    gain.gain.exponentialRampToValueAtTime(0.0001, now + 0.05);


    oscillator.connect(gain);

    gain.connect(ctx.destination);


    oscillator.start(now);

    oscillator.stop(now + 0.055);


    setFlash(true);

    window.setTimeout(() => setFlash(false), 90);

  };


  const stopMetronome = () => {

    if (intervalRef.current) {

      clearInterval(intervalRef.current);

      intervalRef.current = null;

    }

    setIsPlaying(false);

    setActiveSong(null);

  };


  const startMetronome = async (song) => {

    const isSameSongPlaying = activeSong?.title === song.title && isPlaying;


    if (isSameSongPlaying) {

      stopMetronome();

      return;

    }


    if (intervalRef.current) {

      clearInterval(intervalRef.current);

      intervalRef.current = null;

    }


    setActiveSong(song);

    setIsPlaying(true);


    const intervalMs = 60000 / song.bpm;


    await clickSound();


    intervalRef.current = window.setInterval(() => {

      clickSound();

    }, intervalMs);

  };


  useEffect(() => {

    return () => {

      if (intervalRef.current) clearInterval(intervalRef.current);

      if (audioContextRef.current && audioContextRef.current.state !== "closed") {

        audioContextRef.current.close();

      }

    };

  }, []);


  return (

    <div className="min-h-screen bg-black text-white flex items-center justify-center p-4">

      <Card className="w-full max-w-md rounded-3xl shadow-2xl border-0 bg-zinc-900">

        <CardContent className="p-6 md:p-8">

          <div className="flex items-center gap-3 mb-6">

            <div className="w-11 h-11 rounded-2xl bg-white text-black flex items-center justify-center shadow-md">

              <Music className="w-5 h-5" />

            </div>

            <div>

              <h1 className="text-2xl font-bold leading-tight">

                Once A Month - Bei Chéz Heinz

              </h1>

              <p className="text-sm text-zinc-400">Tap = Start / Stop</p>

            </div>

          </div>


          <div className="mb-6 rounded-3xl bg-black text-white p-6 text-center border border-zinc-800">

            <div

              className={`mx-auto mb-4 h-4 w-4 rounded-full transition-all duration-75 ${

                flash ? "scale-125 bg-white" : "scale-100 bg-zinc-600"

              }`}

            />

            <div className="text-xs uppercase tracking-[0.3em] text-zinc-500">Aktiv</div>

            <div className="mt-2 text-3xl font-bold">

              {activeSong ? activeSong.title : "–"}

            </div>

            <div className="mt-1 text-lg text-zinc-400">

              {activeSong ? `${activeSong.bpm} BPM` : "Noch nicht gestartet"}

            </div>

          </div>


          <div className="grid gap-4">

            {SONGS.map((song) => {

              const active = activeSong?.title === song.title && isPlaying;

              return (

                <Button

                  key={song.title}

                  onClick={() => startMetronome(song)}

                  className={`w-full rounded-2xl h-20 text-lg font-bold shadow-lg ${

                    active ? "bg-white text-black" : "bg-zinc-800 text-white"

                  }`}

                >

                  <span className="flex items-center justify-between w-full px-2">

                    <span className="flex items-center gap-3">

                      {active ? <Square className="w-5 h-5" /> : <Play className="w-5 h-5" />}

                      {song.title}

                    </span>

                    <span>{song.bpm}</span>

                  </span>

                </Button>

              );

            })}

          </div>

        </CardContent>

      </Card>

    </div>

  );

}
